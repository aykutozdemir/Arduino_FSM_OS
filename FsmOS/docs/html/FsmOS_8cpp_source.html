60template <typename T>
61bool TimerT<T>::isExpired() const
62{
63 if (durationMs == 0)
64 {
65 return true;
66 }
67
68 T current_time = static_cast<T>(OS.now());
69
70 // Handle timer overflow correctly using safer arithmetic
71 if (current_time >= startMs)
72 {
73 // No overflow case - simple subtraction
74 return (current_time - startMs) >= durationMs;
75 }
76 else
77 {
78 // Timer overflow occurred - calculate elapsed time correctly
79 T max_value = static_cast<T>(~static_cast<T>(0));
80
81 // Calculate elapsed time: time before overflow + time after overflow
82 T elapsed_before_overflow = max_value - startMs + 1;
83 T elapsed_after_overflow = current_time;
84 T total_elapsed = elapsed_before_overflow + elapsed_after_overflow;
85
86 // Check for overflow in the addition
87 if (total_elapsed < elapsed_before_overflow)
88 {
89 // Addition overflowed, timer definitely expired
90 return true;
91 }
92
93 return total_elapsed >= durationMs;
94 }
95}

645static void init_stack_canary()
646{
647 extern char __bss_end;
648 extern char *__brkval;
649 // Current stack pointer approximation via address of a local
650 uint8_t sp_probe;
651 uint8_t *sp = (uint8_t *)&sp_probe;
652
653 // Compute heap top
654 uint8_t *heap_top = (uint8_t *)(__brkval ? __brkval : &__bss_end);
655
656 // Safety margin to keep above the canary (below SP)
657#ifndef FSMOS_STACK_CANARY_MARGIN
658#define FSMOS_STACK_CANARY_MARGIN 32
659#endif
660
661 // Region end just below current SP
662 uint8_t *region_end = sp - FSMOS_STACK_CANARY_MARGIN;
663
664 // Start from heap top - cover entire free RAM area
665 uint8_t *region_start = heap_top;
666
667 if (region_start >= region_end)

669 // Nothing to initialize; leave pointers null
670 g_canary_start = g_canary_end = nullptr;
671 return;
672 }

674 g_canary_start = region_start;
675 g_canary_end = region_end;
676
677 // Disable interrupts around the fill to keep ISR stacks from interfering
678 uint8_t sreg = SREG;
679 cli();
680 for (uint8_t *p = g_canary_start; p <= g_canary_end; ++p)
681 {
682 *p = STACK_CANARY_BYTE;
683 }
684 SREG = sreg; // restore interrupt state
685}

1109Task *Scheduler::findNextTask()
1110{
1111 Task *nextTask = nullptr;
1112 uint16_t shortestTime = UINT16_MAX;
1113
1114 TaskNode *current = taskHead;
1115 while (current != nullptr)
1116 {
1117 if (current->task && current->task->isActive() && current->task->remainingTime == 0)
1118 {
1119 // If task declares a message production budget, ensure enough free queue slots
1120 uint8_t budget = current->task->getMaxMessageBudget();
1121 // Only use default if task hasn't explicitly set a budget (inherited default)
1122 if (budget == DEFAULT_TASK_MESSAGE_BUDGET)
1123 {
1124 // Task is using the default, which is fine
1125 }
1126 // If budget is 0, task explicitly wants no message production
1127 // If budget > 0, task has explicit budget
1128 // In both cases, respect the task's decision
1129 if (getFreeQueueSlots() < budget)
1130 {
1131 // Skip this task for now; not enough capacity to accept its messages
1132 }
1133 else if (nextTask == nullptr)
1134 {
1135 nextTask = current->task;
1136 }
1137 else if (current->task->getPriority() > nextTask->getPriority())
1138 {
1139 // Higher enum value wins (PRIORITY_SYSTEM=7 > PRIORITY_LOWEST=0)
1140 nextTask = current->task;
1141 }
1142 else if (current->task->getPriority() == nextTask->getPriority())
1143 {
1144 // Same priority, smaller task ID wins
1145 if (current->task->getId() < nextTask->getId())
1146 {
1147 nextTask = current->task;
1148 }
1149 }
1150 }
1151 else if (current->task && current->task->isActive() && current->task->remainingTime < shortestTime)
1152 {
1153 shortestTime = current->task->remainingTime;
1154 }
1155 current = current->next;
1156 }
1157
1158 return nextTask;

1189{
1190 // Set scheduled time (when task should have started)
1191 task->scheduledTime = currentTime - task->getPeriod();
1192 task->actualStartTime = currentTime;
1193
1194 // Check for delay
1195 if (task->actualStartTime > task->scheduledTime)

1221void Scheduler::updateTaskStatistics(Task *task, uint32_t execStart)
1222{
1223 // Calculate execution time
1224 uint32_t execTime = micros() - execStart;
1225 uint16_t execTime16 = (execTime > 65535) ? 65535 : static_cast<uint16_t>(execTime);
1226
1227 // Update run count
1228 if (task->runCount < 65535)
1229 {
1230 task->runCount++;
1231 }
1232
1233 // Update max execution time
1234 if (execTime16 > task->maxExecTimeUs)
1235 {
1236 task->maxExecTimeUs = execTime16;
1237 }
1238
1239 // Update average execution time
1240 if (task->runCount == 1)
1241 {
1242 task->avgExecTimeUs = execTime16;
1243 }
1244 else if (task->runCount == 65535)
1245 {
1246 // Exponential moving average to avoid overflow
1247 uint32_t diff = execTime16 - task->avgExecTimeUs;
1248 int32_t adjustment = diff / 1000; // Slow adaptation
1249 int32_t newAvg = task->avgExecTimeUs + adjustment;
1250
1251 // Clamp to valid range
1252 if (newAvg < 0)
1253 {
1254 newAvg = 0;
1255 }
1256 if (newAvg > 65535)
1257 {
1258 newAvg = 65535;
1259 }
1260
1261 task->avgExecTimeUs = static_cast<uint16_t>(newAvg);
1262 }
1263 else
1264 {
1265 // Simple moving average
1266 uint32_t newAvg = ((uint32_t)task->avgExecTimeUs * (task->runCount - 1) + execTime16) / task->runCount;
1267 task->avgExecTimeUs = (newAvg > 65535) ? 65535 : static_cast<uint16_t>(newAvg);
1268 }
1269}

1774{

1776 {
1777 return false;
1778 }
1779
1780 // Allocate up to 4 nodes, but dont exceed hard cap
1781 uint8_t canAdd = static_cast<uint8_t>(MAX_MESSAGE_POOL_SIZE - totalNodes);

1783 for (uint8_t i = 0; i < toAdd; i++)

1797 n->payload.msg.refCount = 0;

1829 slot.msg.arg = arg;
1830 slot.msg.refCount = 0;

1834 {
1835 msgTail->next = node;

1837 }
1838 else
